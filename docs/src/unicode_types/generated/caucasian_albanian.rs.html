    fn into(self) -> u32 {
        let c: char = self.into();

            .escape_unicode()
            .to_string()
            .replace("\\u{", "")
            .replace("}", "");
        u32::from_str_radix(&hex, 16).unwrap()
    }
}

        } else {

        }
    }
