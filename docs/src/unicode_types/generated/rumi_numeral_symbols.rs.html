impl Into<u32> for RumiNumeralSymbols {
    fn into(self) -> u32 {

            .escape_unicode()
            .to_string()
            .replace("\\u{", "")
            .replace("}", "");
        u32::from_str_radix(&hex, 16).unwrap()
    }
}
